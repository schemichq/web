The CLI ships as `schemic`, with `sc` as a short alias. Run any command with `npx schemic <command>`. Commands that talk to a database accept the [global connection flags](#global-flags) and read defaults from `schemic.config.ts`.

## Migrations

<CommandRef
  id="gen"
  name="schemic gen [name]"
  desc="Diff the schema against the snapshot and write a migration for the difference. Writes nothing when there are no changes."
  synopsis="schemic gen add_user_role"
  flags={[
    { flag: "-y", type: "boolean", desc: "Skip the confirmation prompt." },
    { flag: "--baseline", type: "boolean", desc: "Regenerate the whole schema from an empty snapshot as one migration." },
    { flag: "--force", type: "boolean", desc: "With --baseline, replace existing migrations with the fresh baseline." }
  ]}
/>

<CommandRef
  id="migrate"
  name="schemic migrate [count]"
  desc="Apply pending migrations in order — all, the next N, or up to a tag. Alias: up."
  synopsis="schemic migrate --to 20260613090000_add_role"
  flags={[
    { flag: "--to <tag>", type: "string", desc: "Apply up to and including this migration." }
  ]}
/>

<CommandRef
  id="rollback"
  name="schemic rollback [count]"
  desc="Revert applied migrations newest-first. Alias: down."
  synopsis="schemic rollback 2"
  flags={[
    { flag: "--to <tag>", type: "string", desc: "Revert everything applied after this tag." }
  ]}
/>

<CommandRef
  id="status"
  name="schemic status"
  desc="Show applied vs pending migrations, flagging files that changed after apply or are missing."
  synopsis="schemic status --json"
  flags={[
    { flag: "--json", type: "boolean", desc: "Output the status as JSON." }
  ]}
/>

<CommandRef
  id="new"
  name="schemic new <kind> <name>"
  desc="Scaffold a starter schema file for an entity at <schema>/<kind-folder>/<name>.ts — a realistic starter, not a bare stub. Surreal kinds map to folders: table → tables/, relation → relations/, view → views/, function → functions/, access → access/, event → events/, analyzer → analyzers/. (index and field are authored inline on a table, not standalone.) Requires a directory schema layout; never overwrites an existing file."
  synopsis="schemic new table user"
  flags={[]}
/>

<CommandRef
  id="check"
  name="schemic check"
  desc="Replay migrations in a throwaway engine and confirm they reproduce the schema. Exits non-zero on mismatch."
  synopsis="schemic check --schema"
  flags={[
    { flag: "--schema", type: "boolean", desc: "Check the schema specifically." }
  ]}
/>

<CommandRef
  id="snapshot-reset"
  name="schemic snapshot reset"
  desc="Clear the migration snapshot so the next schemic gen baselines the full schema."
  synopsis="schemic snapshot reset"
  flags={[]}
/>

<CommandRef
  id="unlock"
  name="schemic unlock"
  desc="Release a stale migration lock left by an interrupted run."
  synopsis="schemic unlock"
  flags={[]}

## Live database

<CommandRef
  id="diff"
  name="schemic diff"
  desc="Preview the pending change without writing a migration. Diffs against the snapshot, or a live database with --live."
  synopsis="schemic diff --live --ts"
  flags={[
    { flag: "--live", type: "boolean", desc: "Diff against the running database." },
    { flag: "--ts", type: "boolean", desc: "Render the delta as TypeScript instead of SurrealQL." },
    { flag: "--down", type: "boolean", desc: "Show the rollback (reverse) direction." },
    { flag: "--full", type: "boolean", desc: "Show complete definitions, not only the delta." },
    { flag: "--inline", type: "boolean", desc: "Inline format." },
    { flag: "-p, --patch", type: "boolean", desc: "Patch-style output." },
    { flag: "--pager", type: "boolean", desc: "Page long output." },
    { flag: "--json", type: "boolean", desc: "Machine-readable output." },
    { flag: "--watch", type: "boolean", desc: "Re-run as the schema changes." }
  ]}
/>

<CommandRef
  id="push"
  name="schemic push"
  desc="Apply the schema directly to the database, no migration files. Alias: sync. Prunes stale objects by default."
  synopsis="schemic push --dry-run"
  flags={[
    { flag: "--dry-run", type: "boolean", desc: "Show what would change without applying." },
    { flag: "--no-prune", type: "boolean", desc: "Apply additions and changes only; keep stale objects." },
    { flag: "--watch", type: "boolean", desc: "Re-push as the schema changes." }
  ]}
/>

<CommandRef
  id="pull"
  name="schemic pull"
  desc="Introspect the database and update your s.* files to match. Previews by default; non-destructive."
  synopsis="schemic pull --write"
  flags={[
    { flag: "--write", type: "boolean", desc: "Apply the preview to your .ts files." },
    { flag: "--merge", type: "boolean", desc: "Keep local-only entities; apply the database's changes alongside them." },
    { flag: "--discard", type: "boolean", desc: "Mirror the database exactly, dropping local-only definitions." }
  ]}
/>

<CommandRef
  id="seed"
  name="schemic seed [name]"
  desc="Run the project's seeds against a connection. Seeds live in database/seed/: bare schemic seed runs database/seed/index.ts (or every seed in filename order if there is none), a name runs one seed, and --all runs them all. A flat database/seed.ts still works as a single seed."
  synopsis="schemic seed [name] [--all]"
  flags={[
    { flag: "[name]", type: "string", desc: "Run the single named seed (its filename without the NN- ordering prefix)." },
    { flag: "--all", type: "boolean", desc: "Run every seed in database/seed/ in filename order, even when an index.ts is present." }
  ]}
/>

## Global flags

These apply to every command that connects to a database. Each overrides, for that command, the resolved connection from `schemic.config.ts`.

| Flag | Description |
|------|-------------|
| `--url <url>` | Database endpoint, e.g. `ws://localhost:8000`. |
| `--namespace <ns>` | Target namespace. |
| `--database <db>` | Target database. |
| `--username <user>` | Auth username. |
| `--password <pass>` | Auth password. |
| `--auth-level <level>` | Sign-in level: `root`, `namespace`, or `database`. |
| `-c, --config <path>` | Path to the config file. |
| `--[no-]tables` | Include or exclude tables from the operation. |
| `--[no-]functions` | Include or exclude functions. |
| `--[no-]access` | Include or exclude access definitions. |
